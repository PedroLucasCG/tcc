    {\largefont \textbf{LISTA DE SIGLAS E ACRÔNIMOS}}

    \vspace{0.5cm}

    \justifying
    \begin{enumerate}
        \item ADS – \textit{Análise e Desenvolvimento de Sistemas}
        \item CAP – \textit{Consistency, Availability, Partition Tolerance} – Consistência, Disponibilidade e Tolerância à Partição
        \item CmRDT – \textit{Commutative Replicated Data Type} – Tipo de Dado Replicado Baseado em Operações
        \item CRDT – \textit{Conflict-free Replicated Data Type} – Tipo de Dado Replicado Sem Conflito
        \item CvRDT – \textit{Convergent Replicated Data Type} – Tipo de Dado Replicado Baseado em Estado
        \item HTTP – \textit{HyperText Transfer Protocol} – Protocolo de Transferência de Hipertexto
        \item IPFS – \textit{InterPlanetary File System} – Sistema de Arquivos Interplanetário
        \item OT – \textit{Operational Transformation} – Transformação Operacional
        \item P2P – \textit{Peer-to-Peer} – Rede de Pares
        \item RAID – \textit{Redundant Array of Independent Disks} – Conjunto Redundante de Discos Independentes
        \item RFC – \textit{Request for Comments} \\ – Solicitação de Comentários (Documentos de Padronização)
        \item SQL – \textit{Structured Query Language}\\ – Linguagem de Consulta Estruturada
        \item TCC – \textit{Trabalho de Conclusão de Curso}
        \item TCP/IP – \textit{Transmission Control Protocol / Internet Protocol} – Protocolo de Controle de Transmissão / Protocolo de Internet
    \end{enumerate}

\end{titlepage}
